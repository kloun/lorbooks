мЮГЮДцКЮБЮ 6. сЯРЮМНБЙЮ НЯМНБМШУ ЙНЛОНМЕМРНБ ЯХЯРЕЛШбОЕПЕД

яНГДЮМХЕ ДХПЕЙРНПХИ

дЮБЮИРЕ ЯНГДЮДХЛ ЯРПСЙРСПС ТЮИКНБНИ ЯХЯРЕЛШ LFS. мЮВМЕЛ Я ЯНГДЮМХЪ АНКЕЕ ХКХ ЛЕМЕЕ ЯРЮМДЮПРМНЦН ДЕПЕБЮ Я ОНЛНЫЭЧ ЯКЕДСЧЫХУ ЙНЛЮМД:

mkdir -p /{bin,boot,dev/pts,etc/opt,home,lib,mnt,proc} &&
mkdir -p /{root,sbin,tmp,usr/local,var,opt} &&
for dirname in /usr /usr/local
    do
    mkdir $dirname/{bin,etc,include,lib,sbin,share,src}
    ln -s share/{man,doc,info} $dirname
    mkdir $dirname/share/{dict,doc,info,locale,man}
    mkdir $dirname/share/{nls,misc,terminfo,zoneinfo}
    mkdir $dirname/share/man/man{1,2,3,4,5,6,7,8}
done &&
mkdir /var/{lock,log,mail,run,spool} &&
mkdir -p /var/{tmp,opt,cache,lib/misc,local} &&
mkdir /opt/{bin,doc,include,info} &&
mkdir -p /opt/{lib,man/man{1,2,3,4,5,6,7,8}} &&
ln -s ../var/tmp /usr

оН СЛНКВЮМХЧ ДХПЕЙРНПХХ ЯНГДЮЧРЯЪ Я ОПЮБЮЛХ ДНЯРСОЮ 755, НДМЮЙН ЩРН ОНДУНДХР МЕ ДКЪ БЯЕУ ДХПЕЙРНПХИ. лШ ЯДЕКЮЕЛ ДБЮ ХГЛЕМЕМХЪ: ДКЪ ДНЛЮЬМЕЦН ЙЮРЮКНЦЮ root Х ДКЪ ЙЮРЮКНЦЮ БПЕЛЕММШУ ТЮИКНБ.

chmod 0750 /root &&
chmod 1777 /tmp /var/tmp

оЕПБНЕ ХГЛЕМЕМХЕ ЦЮПЮМРХПСЕР, ВРН МЕ БЯЕ ОНКЭГНБЮРЕКХ ХЛЕЧР ДНЯРСО Й ДХПЕЙРНПХХ /root directory -- ЮМЮКНЦХВМШЕ ДЕИЯРБХЪ БШОНКМЪЕР НАШВМШИ ОНКЭГНБЮРЕКЭ ЯН ЯБНХЛ ДНЛЮЬМХЛ ЙЮРЮКНЦНЛ. бРНПНЕ ХГЛЕМЕМХЕ ДЮЕР КЧАНЛС ОНКЭГНБЮРЕКЧ ОПЮБН МЮ ГЮОХЯЭ Б ДХПЕЙРНПХХ /tmp Х /var/tmp, НДМЮЙН МЕ ПЮГПЕЬЮЕР СДЮКЪРЭ ХГ МХУ ТЮИКШ ДПСЦХУ ОНКЭГНБЮРЕКЕИ. сДЮКЕМХЕ ВСФХУ ТЮИКНБ НОПЕДЕКЪЕРЯЪ ОЮПЮЛЕРПНЛ "sticky bit" -- МЮХБШЯЬХИ ДБНХВМШИ ГМЮЙ Б ДБНХВМНИ ЛЮЯЙЕ 1777.

хРЮЙ, ОНЯКЕ ЯНГДЮМХЪ ДХПЕЙРНПХИ ОЕПЕЛЕЯРХРЕ tar-ЮПУХБШ ХЯУНДМШУ ЙНДНБ ОЮЙЕРНБ, ГЮЦПСФЕММШУ ОН ХМЯРПСЙЖХЪЛ цКЮБШ 3, Б МЕЙНРНПШИ ЯЮЛНЯРНЪРЕКЭМН ЯНГДЮММШИ ОНДЙЮРЮКНЦ Б /usr/src.

яННРБЕРЯРБХЕ FHS

оПХ ЯНГДЮМХХ ДХПЕЙРНПХИ ЛШ НЯМНБШБЮКХЯЭ МЮ ЯРЮМДЮПРЕ FHS (ЯЛ. http://www.pathname.com/fhs/). оНЛХЛН ЯНГДЮММШУ ЙЮРЮКНЦНБ ДЮММШИ ЯРЮМДЮПР ЯРЮБХР СЯКНБХЕЛ МЮКХВХЕ ЙЮРЮКНЦНБ /usr/local/games Х /usr/share/games, МН, ОН МЮЬЕЛС ЛМЕМХЧ, ДКЪ АЮГНБНИ ЯХЯРЕЛШ НМХ МЕ МСФМШ. нДМЮЙН ЛНФЕРЕ ЯЛЕКН ОПХБЕЯРХ ЯБНЧ ЯХЯРЕЛС Й ОНКМНЛС ЯННРБЕРЯРБХЧ Я FHS. вРН ЙЮЯЮЕРЯЪ ЯРПСЙРСПШ ЙЮРЮКНЦЮ /usr/local/share, ЯРЮМДЮПР FHS ЯРПНЦН МЕ НЦНБЮПХБЮЕР  ЕЦН ЯНДЕПФХЛНЕ, РЮЙ ВРН ЛШ ПЕЬХКХ ЯНГДЮРЭ Б МЕЛ МЕНАУНДХЛШЕ, ОН МЮЬЕЛС САЕФДЕМХЧ, ОНДЙЮРЮКНЦХ.
